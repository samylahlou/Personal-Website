\begin{definition}
    The Factorial Function $\Pi : \C \setminus \{-1, -2, -3, ...\} \to \C$ is defined by
    $$\Pi(z) = \int_{0}^{\infty}e^{-t}t^z dt.$$
\end{definition}

\begin{definition}
    The Gamma Function $\Gamma : \C \setminus \{-1, -2, -3, ...\} \to \C$ is defined by
    $$\Gamma(z) = \int_{0}^{\infty}e^{-t}t^{z-1}dt.$$
\end{definition}
